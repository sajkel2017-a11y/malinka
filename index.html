MAIN PROJECT Python Raspberry Pi

System Grzewczy + PIR (Światło) + Wybór Regionu

Pełna logika: Histereza temperatury, obsługa czujnika ruchu (PIR) przez callback, przeliczanie Kelwinów/Celsiuszy oraz walidacja progów.

import time
import pigpio
import glob

# --- KONFIGURACJA PINÓW ---
HEATER_PIN = 17
LIGHT_PIN = 27
PIR_PIN = 22

# Inicjalizacja pigpio
pi = pigpio.pi()
pi.set_mode(HEATER_PIN, pigpio.OUTPUT)
pi.set_mode(LIGHT_PIN, pigpio.OUTPUT)
pi.set_mode(PIR_PIN, pigpio.INPUT)

# Znalezienie czujnika 1-Wire
base_dir = '/sys/bus/w1/devices/'
device_folder = glob.glob(base_dir + '28*')[0]
device_file = device_folder + '/w1_slave'

# --- FUNKCJE ---
def get_temp():
    with open(device_file, 'r') as f:
        content = f.readlines()
    if "YES" in content[0]:
        return float(content[1].split("t=")[1]) / 1000.0
    return None

def pir_callback(gpio, level, tick):
    if level == 1:  # Ruch wykryty
        print("Wykryto ruch! Włączam światło na 10 sekund.")
        pi.write(LIGHT_PIN, 1)
        time.sleep(10)
        pi.write(LIGHT_PIN, 0)

# Ustawienie callbacku dla czujnika PIR
pi.callback(PIR_PIN, pigpio.RISING_EDGE, pir_callback)

# --- START PROGRAMU ---
print("--- KONFIGURACJA SYSTEMU ---")
region = input("Wybierz region (EU/TECH): ").upper()
t_on = float(input("Podaj próg włączenia ogrzewania: "))
t_off = float(input("Podaj próg wyłączenia ogrzewania: "))

# Walidacja danych
if t_on >= t_off:
    print("BŁĄD: Próg włączenia musi być niższy niż wyłączenia!")
    exit()

# Przeliczenie dla Kelwinów
offset = 273.15 if region == "TECH" else 0
t_on += offset
t_off += offset
jednostka = "K" if region == "TECH" else "°C"

print(f"System uruchomiony. Region: {region}. Zakres: {t_on:.2f} - {t_off:.2f} {jednostka}")

try:
    while True:
        temp = get_temp()
        if temp is not None:
            # Konwersja na Kelwiny jeśli region TECH
            if region == "TECH": temp += 273.15
            print(f"Temperatura: {temp:.2f} {jednostka}")

            # Logika histerezy
            if temp < t_on:
                pi.write(HEATER_PIN, 1) # Grzej
            elif temp > t_off:
                pi.write(HEATER_PIN, 0) # Przestań grzać
        
        time.sleep(10) # Pomiar co 10 sekund

except KeyboardInterrupt:
    print("\nZamykanie systemu...")
finally:
    pi.write(HEATER_PIN, 0)
    pi.write(LIGHT_PIN, 0)
    pi.stop()
